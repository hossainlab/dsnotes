Drop Missing Values¶

# shape 
ufo.shape 

(18241, 5)

# drop missing: drop row contains missing values 
# it is inplace = False 
ufo.dropna(how='any').shape

(2486, 5)

# how=all 
ufo.dropna(how='all').shape 

(18241, 5)

# subset: any 
ufo.dropna(subset=['City', 'Shape Reported'], how='any').shape

(15576, 5)

# subset: all 
ufo.dropna(subset=['City', 'Shape Reported'], how='all').shape

(18237, 5)
